Using a SMTP server called Bamboo
Set Mailer = Server.CreateObject ("Bamboo.SMTP")
Mailer.Server = "mailhub.yourhost.co.uk"
Mailer.FromName = name
Mailer.Rcpt = "info@yourhost.co.uk"
Mailer.Subject = subject
BodyText = name & "(" & email & ")" & " has just said this - " & body
Mailer.Message = BodyText
Mailer.Send

Here is an example using JMail Set JMail = Server.CreateObject("JMail.SMTPMail")
JMail.ServerAddress = " mailhub.yourhost.co.uk"
JMail.Sender = email JMail.Subject = subject
'No equals sign!!
JMail.AddRecipient "info@yourhost.co.uk"
JMail.Body = name & "(" & email & ")" & " has just said this - " & body
' 1 - highest priority (Urgent)
' 3 - normal
' 5 - lowest
JMail.Priority = 3
JMail.AddHeader "Originating-IP", Request.ServerVariables("REMOTE_ADDR")
JMail.Execute

Sending mail with ASPMail
Set Mailer = Server.CreateObject("SMTPsvg.Mailer")
